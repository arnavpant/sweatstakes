**Step 4 — Replace `src/app/layout.tsx` with root layout:**

```typescript
import type { Metadata, Viewport } from 'next'
import { Plus_Jakarta_Sans } from 'next/font/google'
import './globals.css'

const jakartaSans = Plus_Jakarta_Sans({
  subsets: ['latin'],
  weight: ['400', '700'],
  variable: '--font-sans',
})

export const viewport: Viewport = {
  width: 'device-width',
  initialScale: 1,
  maximumScale: 1,
  viewportFit: 'cover',
  themeColor: '#001233',
}

export const metadata: Metadata = {
  title: 'SweatStakes',
  description: 'Hold your friends accountable to their fitness goals',
  appleWebApp: {
    capable: true,
    statusBarStyle: 'black-translucent',
    title: 'SweatStakes',
  },
}

export default function RootLayout({
  children,
}: {
  children: React.ReactNode
}) {
  return (
    <html lang="en" className={jakartaSans.variable}>
      <body className="bg-background text-on-surface font-sans antialiased min-h-dvh">
        <div className="pt-[env(safe-area-inset-top)]">
          {children}
        </div>
      </body>
    </html>
  )
}
```

Key details:
- `Plus_Jakarta_Sans` loaded with weights `['400', '700']` only (per UI-SPEC Typography section)
- `viewportFit: 'cover'` enables `env(safe-area-inset-*)` on iOS (per D-21)
- `maximumScale: 1` prevents accidental zoom on mobile form inputs
- `appleWebApp` metadata for iOS PWA (per D-20)
- Body has `bg-background text-on-surface font-sans antialiased min-h-dvh`
- Safe area top padding via `pt-[env(safe-area-inset-top)]` wrapper div (per D-21)
